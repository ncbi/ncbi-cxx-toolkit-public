    NCBI_set_test_command(seqdb_perf -db pataa -dbtype prot -scan_uncompressed -num_threads 1)
  NCBI_end_test()

    NCBI_set_test_command(seqdb_perf -db pataa -dbtype prot -get_metadata)
  NCBI_end_test()
